Thread-safety pattern for group handlers (from RESEARCH.md):
```python
class CreateGroupHandler(CommandHandler):
    def __init__(self, groups: dict, event_bus: EventBus):
        self._groups = groups
        self._event_bus = event_bus
        self._lock = threading.Lock()

    def handle(self, command):
        with self._lock:
            if command.group_id in self._groups:
                raise ValidationError(f"Group already exists: {command.group_id}")
            group = ProviderGroup(...)
            self._groups[command.group_id] = group
        # Publish event OUTSIDE lock (no I/O under lock)
        self._event_bus.publish(GroupCreated(...))
        return {"group_id": ..., "created": True}
```

DeleteGroupHandler pattern — stop members OUTSIDE lock:
```python
class DeleteGroupHandler(CommandHandler):
    def handle(self, command):
        with self._lock:
            group = self._groups.get(command.group_id)
            if group is None:
                raise ProviderNotFoundError(command.group_id)
            del self._groups[command.group_id]
        # I/O (stop) outside lock
        group.stop_all()
        for event in group.collect_events():
            self._event_bus.publish(event)
        self._event_bus.publish(GroupDeleted(group_id=command.group_id))
        return {"group_id": command.group_id, "deleted": True}
```

AddGroupMemberHandler pattern — needs both repository and groups:
```python
class AddGroupMemberHandler(CommandHandler):
    def __init__(self, repository: IProviderRepository, groups: dict, event_bus: EventBus):
        self._repository = repository
        self._groups = groups
        self._event_bus = event_bus
        self._lock = threading.Lock()

    def handle(self, command):
        provider = self._repository.get(command.member_id)
        if provider is None:
            raise ProviderNotFoundError(command.member_id)
        with self._lock:
            group = self._groups.get(command.group_id)
            if group is None:
                raise ProviderNotFoundError(command.group_id)
            group.add_member(provider, weight=command.weight, priority=command.priority)
        for event in group.collect_events():
            self._event_bus.publish(event)
        return {"group_id": command.group_id, "member_id": command.member_id, "added": True}
```

    ```python
    def update(
        self,
        strategy: str | None = None,
        description: str | None = None,
        min_healthy: int | None = None,
    ) -> None:
        """Update mutable configuration fields under self._lock.

        Only non-None arguments are applied. Records GroupUpdated event.
        """
        with self._lock:
            if strategy is not None:
                self._strategy = LoadBalancerStrategy(strategy)
            if description is not None:
                self._description = description
            if min_healthy is not None:
                self._min_healthy = max(1, min_healthy)
        self._record_event(GroupUpdated(group_id=self.id))
    ```
