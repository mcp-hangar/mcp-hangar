    ```python
    class DockerNetworkMonitor:
        def __init__(self) -> None:
            # Guard import
            import docker
            self._client = docker.from_env()
            self._ss_available: dict[str, bool] = {}  # container_id -> ss works

        def poll_connections(self, provider_id: str) -> list[NetworkObservation]:
            # Find container by label filter
            containers = self._client.containers.list(
                filters={"label": f"mcp-hangar.provider-id={provider_id}", "status": "running"}
            )
            if not containers:
                return []
            container = containers[0]
            connections = self._get_connections(container)
            now = time.time()
            return [
                NetworkObservation(
                    timestamp=now, provider_id=provider_id,
                    destination_host=host, destination_port=port,
                    protocol=protocol, direction="outbound",
                )
                for host, port, protocol in connections
            ]

        def _get_connections(self, container) -> list[tuple[str, int, str]]:
            cid = container.short_id
            # Try ss if not known to be unavailable
            if self._ss_available.get(cid, True):
                try:
                    exit_code, output = container.exec_run("ss -tnp", demux=True)
                    if exit_code == 0 and output[0]:
                        self._ss_available[cid] = True
                        return parse_ss_output(output[0].decode("utf-8"))
                    else:
                        self._ss_available[cid] = False
                except Exception:
                    self._ss_available[cid] = False
            # Fallback to /proc/net/tcp
            try:
                exit_code, output = container.exec_run("cat /proc/net/tcp", demux=True)
                if exit_code == 0 and output[0]:
                    return parse_proc_net_tcp(output[0].decode("utf-8"))
            except Exception:
                pass
            return []
    ```

    Guard top-level import: `try: import docker; DOCKER_AVAILABLE = True; except ImportError: DOCKER_AVAILABLE = False`
